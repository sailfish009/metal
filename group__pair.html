§ first

template<typename seq >
using metal::first = typedef metal::if_<metal::is_pair<seq>, metal::front<seq>>

#include <metal/pair/first.hpp>

Description

Retrieves the first element of a Pair .

Usage

For any Pair p

using result = metal::first<p>;
Returns
: Value
Semantics:
If p contains Values p_0 and p_1 in that order, then
using result = p_0;

Example

IS_SAME(metal::first<metal::pair<void, void*>>, void);

§ is_pair

template<typename val >
using metal::is_pair = typedef typename detail::_is_pair<val>::type

#include <metal/pair/pair.hpp>

Description

Checks whether some Value is a Pair .

Usage

For any Value val

using result = metal::is_pair<val>;
Returns
: Number
Semantics:
If val is a Pair , then
using result = metal::true_;
otherwise
using result = metal::false_;

Example

IS_SAME(metal::is_pair<metal::pair<int, unsigned>>, metal::true_);
IS_SAME(metal::is_pair<metal::list<void, void*>>, metal::true_);
IS_SAME(metal::is_pair<metal::list<void>>, metal::false_);
